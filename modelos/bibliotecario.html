# ========================================== # CLASE: Bibliotecario (hereda de Persona) # Es el usuario que inicia sesión en el sistema. # ========================================== from modelos.persona import Persona class Bibliotecario(Persona): def __init__(self, id_persona, nombre, usuario, contrasena): super().__init__(id_persona, nombre) # Atributos privados propios de Bibliotecario (encapsulamiento) self.__usuario = usuario self.__contrasena = contrasena # Getters def get_usuario(self): return self.__usuario # Setters def set_contrasena(self, contrasena): self.__contrasena = contrasena # Verifica si la contraseña ingresada es correcta def verificar_contrasena(self, contrasena): return self.__contrasena == contrasena # Polimorfismo: se sobrescribe el método de la clase padre def mostrar_datos(self): datos_base = super().mostrar_datos() return f"{datos_base} | Usuario: {self.__usuario} | Cargo: Bibliotecario"
